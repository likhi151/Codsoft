import java.util.Random;
import java.util.Scanner;

public class NumberGuessingGame {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();

        int maxAttempts = 10;
        int rangeStart = 1;
        int rangeEnd = 100;
        int score = 0;
        int rounds = 0;

        while (true) {
            rounds++;
            int attempts = 0;
            int numberToGuess = random.nextInt(rangeEnd - rangeStart + 1) + rangeStart;
            System.out.println("\nRound " + rounds);
            System.out.println("Guess the number between " + rangeStart + " and " + rangeEnd + ". You have " + maxAttempts + " attempts.");

            while (attempts < maxAttempts) {
                System.out.print("Enter your guessed number: ");
                int guess = scanner.nextInt();
                attempts++;

                if (guess < numberToGuess) {
                    System.out.println("Too low!");
                } else if (guess > numberToGuess) {
                    System.out.println("Too high!");
                } else {
                    System.out.println("Great! You've guessed the right number.");
                    score += maxAttempts - attempts + 1;
                    break;
                }
            }

            if (attempts == maxAttempts) {
                System.out.println("Oops, you've used all your attempts. The number was " + numberToGuess + ".");
            }

            System.out.println("Score: " + score);

            System.out.print("Do you want to play another round? (yes/no): ");
            String playAgain = scanner.next().toLowerCase();
            if (!playAgain.equals("yes")) {
                break;
            }
        }

        System.out.println("\nFinal Score: " + score);
        System.out.println("Thank you for playing!");

        scanner.close();
    }
}
